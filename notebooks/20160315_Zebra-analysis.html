Run maxeval vs hmax experiments¶

In [4]:
def experiment(M, maxevals, hmaxs):
    rows = []
    for maxeval in maxevals:
        for hmax in hmaxs:
            for i in range(M):
                heur = ShootAndGo(Zebra3(6), maxeval=maxeval, hmax=hmax)
                row = heur.search()
                row['maxeval'] = maxeval
                row['hmax'] = hmax
                rows.append(row)
    return pd.DataFrame(rows)
